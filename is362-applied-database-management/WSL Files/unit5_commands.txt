CREATE TABLE department (
    DepartmentID VARCHAR(3) PRIMARY KEY,
    Name VARCHAR(50),
    AisleStart INT,
    AisleEnd INT,
    RowStart INT,
    RowEnd INT
);

CREATE TABLE works (
    EmployeeID VARCHAR(7) PRIMARY KEY,
    DepartmentID INT,
    Date DATE NOT NULL
);

CREATE TABLE location (
    LocationID VARCHAR(7) PRIMARY KEY,
    Aisle VARCHAR(2),
    Row INT,
    ProductID VARCHAR(7)
);

CREATE TABLE employee (
    EmployeeID VARCHAR(7) PRIMARY KEY,
    FirstName CHAR(25),
    LastName CHAR(25),
    Position VARCHAR(30),
    Phone VARCHAR(11),
    Address VARCHAR(50),
    City VARCHAR(25),
    State CHAR(2),
    Wage FLOAT
);

CREATE TABLE product (
    ProductID VARCHAR(7) PRIMARY KEY AUTO_INCREMENT,
    Location VARCHAR(7),
    Name VARCHAR(50),
    Description CHAR(255),
    Price FLOAT NOT NULL,
    Quantity INT
);

CREATE TABLE transaction (
    TransactionID INT PRIMARY KEY AUTO_INCREMENT,
    Total FLOAT,
    SalesPerson INT,
    PurchaseDate DATE NOT NULL,
    PurchaseTime DATETIME NOT NULL
);
    
CREATE TABLE purchase (
    PurchaseID INT PRIMARY KEY AUTO_INCREMENT,
    CustomerID VARCHAR(7),
    TransactionID INT NOT NULL,
    ProductID VARCHAR(7) NOT NULL,
    Quantity INT,
    Price FLOAT,
);

CREATE TABLE history (
    CustomerID VARCHAR(7) PRIMARY KEY,
    TransactionID INT NOT NULL,
    PurchaseDate DATE,
    Total FLOAT,
    FirstName CHAR(25),
    LastName CHAR(30)
);

CREATE TABLE supply (
    SupplyID INT PRIMARY KEY AUTO_INCREMENT,
    SupplierID VARCHAR(7),
    ProductID VARCHAR(7),
    Quantity INT,
    UnitPrice FLOAT NOT NULL
);

CREATE TABLE customer (
    CustomerID VARCHAR(7) PRIMARY KEY,
    FirstName CHAR(25),
    LastName CHAR(30),
    Email VARCHAR(50),
    Phone VARCHAR(12),
    Address VARCHAR(50),
    City VARCHAR(25),
    State CHAR(2)
);

CREATE TABLE supplier (
    SupplierID VARCHAR(7) PRIMARY KEY,
    Name VARCHAR(50),
    Phone VARCHAR(12)
    Address VARCHAR(50),
    City VARCHAR(25),
    State CHAR(2),
    Details CHAR(255)
);
